	if(#YouUpSeed == 0) goto UpSeed;

		if(#YouUpWater == 0) goto NoUpWater;

			if(#YouUpWater != gettime(5)) goto NoUpWater;

UpSeed:

sleep2 3000;
if(countitem(19110) >= 1){ set #YouSeed,19110; mes "^0000FF มีน้ำยา "+getitemname(19110)+" อยู่"; next; goto UploadPlant; }
if(countitem(19111) >= 1){ set #YouSeed,19111; mes "^0000FF มีน้ำยา "+getitemname(19111)+" อยู่"; next; goto UploadPlant; }
if(countitem(19112) >= 1){ set #YouSeed,19112; mes "^0000FF มีน้ำยา "+getitemname(19112)+" อยู่"; next; goto UploadPlant; }
if(countitem(19113) >= 1){ set #YouSeed,19113; mes "^0000FF มีน้ำยา "+getitemname(19113)+" อยู่"; next; goto UploadPlant; }

goto Main;

UpPlant:
